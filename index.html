const express = require('express'); const cors = require('cors'); const axios = require('axios'); const app = express(); app.use(cors()); app.use(express.json()); const BASE_URL = 'https://draw.ar-lottery01.com/WinGo'; // Global variables let currentGame = null; let gameHistory = []; // ==================== FETCH FUNCTIONS ==================== // Fetch Current Game State async function fetchGameState() { try { const res = await axios.get(`${BASE_URL}/1M.json?ts=${Date.now()}`, { headers: { 'User-Agent': 'Mozilla/5.0 (Linux; Android 10; K) AppleWebKit/537.36' } }); currentGame = res.data; console.log(`✅ Current Period: ${currentGame?.issueNumber}`); return currentGame; } catch (error) { console.error("Game State Error:", error.message); return null; } } // Fetch Latest History async function fetchHistory() { try { const res = await axios.get(`${BASE_URL}/GetHistory`, { headers: { 'User-Agent': 'Mozilla/5.0 (Linux; Android 10; K) AppleWebKit/537.36' } }); if (res.data?.data?.list) { gameHistory = res.data.data.list.map(item => ({ issueNumber: item.issueNumber, number: parseInt(item.number), color: item.color, premium: item.premium })); console.log(`📜 History Updated: ${gameHistory.length} records`); } return gameHistory; } catch (error) { console.error("History Error:", error.message); return []; } } // ==================== ROUTES ==================== app.get('/', (req, res) => { res.json({ status: 'Vanraj WinGo 1M Backend', online: true, currentPeriod: currentGame?.issueNumber, time: new Date().toISOString() }); }); // Get Current Game app.get('/api/game/current', async (req, res) => { if (!currentGame) await fetchGameState(); const now = Date.now(); const endTime = new Date(currentGame?.endTime).getTime(); const remainingSeconds = Math.max(0, Math.floor((endTime - now) / 1000)); res.json({ success: true, current: currentGame, remainingSeconds, isOpen: remainingSeconds > 5, note: "WinGo 1M Live" }); }); // Get History app.get('/api/game/history', (req, res) => { res.json({ success: true, history: gameHistory.slice(0, 15) // Last 15 results }); }); // Place Bet app.post('/api/game/bet', (req, res) => { const { userId, amount, color, number } = req.body; if (!userId || !amount || amount <= 0) { return res.status(400).json({ success: false, msg: "Invalid bet data" }); } res.json({ success: true, betId: `BET_${Date.now()}`, period: currentGame?.issueNumber, amount, color, number, message: "Bet placed successfully!" }); }); // Get Latest Result app.get('/api/game/result', async (req, res) => { await fetchGameState(); await fetchHistory(); res.json({ success: true, latest: gameHistory[0] || null, note: "Official WinGo 1M Result" }); }); // ==================== AUTO REFRESH ==================== setInterval(() => { fetchGameState(); fetchHistory(); }, 3000); // Every 3 seconds // Start Server const PORT = process.env.PORT || 3000; app.listen(PORT, async () => { console.log(`🚀 Vanraj WinGo 1M Server Running on port ${PORT}`); await fetchGameState(); await fetchHistory(); });
